Template operators:
  $type
  $required
  $permit other values (only if this is a collection)
  $path
  $required child attributes

typedef timestamped-object:
  $collection: true
  $permit other values: false
  time stamp:
    $type datetime
    $required true
    $path: this.timestamp
  payload:
    $type anything
    $required optional
    $path: this.payload

typedef taglist
  $collection: true
  $array: true
  $each:
    $type: string
  $each_validates_uniqueness: true

typedef list of list of integers
  $collection: true
  $array: rue
  $each:
    $type:
      $array: true


typedef meal
  time:
    $type: datetime
  name:
